Function to Calculate Returns & Volatility
calc_returns_volatility <- function(stock, ticker) {
  stock %>%
    arrange(Date) %>%
    mutate(
      log_return = log(`Adj Close` / lag(`Adj Close`)),
      vol_30d = rollapply(log_return, 30, sd, fill = NA, align = "right"),
      symbol = ticker
    )
}
